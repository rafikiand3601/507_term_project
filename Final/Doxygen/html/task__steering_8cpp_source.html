59 /** @brief This method is called to run the servo control loop.
60  * @details This function works within the FreeRTOS framework. Once it is called, it
61  * loops through a switch for as long as the main program is running in a finite state
62  * machine. This task uses the shared p_servo_pos variable calculate and set the
63  * PWM signal which activates the servo.
64  */
65 
66 void task_steering::run (void)
67 {
68 
69  // This is an infinite loop; it runs until the power is turned off. There is one
70  // such loop inside the code for each task
71  for (;;)
72  {
73  // Run the finite state machine. The variable 'state' is kept by parent class
74  switch (state)
75  {
76  // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
77  // In state 0, setup the output pin for the servo on PB4(OC2A)
78  case (0):
79  // Need a wave with period ~ 20 ms and pulse length of 1.0-2.0 ms
80  // f = f_clk / N*(1 + TOP)
81  // Want f = 50 [hz]
82  // TOP = 0xFF = 255, f_clk = 16 [Mhz]
83  // N = 1024 ---> f = 61.3 [hz] close enough
84 
85  // Set PB4 as output pin
86  DDRB |= (1 << PB4);
87  // Setup timer register for fast pwm, non-inverting
88  // WGM: fast pwm COM: non-inverting output
89  TCCR2A |= (1 << WGM20) | (1 << WGM21) | (1 << COM2A1);
90  // WGM: fast pwm CS: prescaler = 1024
91  TCCR2B |= (1 << CS22) | (1 << CS21) | (1 << CS20);
92 
93  // Setup of OCR
94  // 1.5 [ms] ----> 667 [hz] ---> 24
95  // Should run 61.3 [hz] period with 1.5 [ms] pulses
96  OCR2A = 24;
97 
98  state = 1;
99  break; // End of state 0
100 
101  // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
102  // In state 1, control the servo position
103  case (1):
104  // Vary OCR to change pulse length.
105  // Pulses are between 1.0 and 2.0 [ms]
106  OCR2A = calc_pwm(p_servo_pos->get ());
107 
108  break; // End of state 1
109 
110  // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
111  // We should never get to the default state. If we do, complain and restart
112  default:
113  *p_serial << PMS ("Illegal state! Resetting AVR") << endl;
114  wdt_enable (WDTO_120MS);
115  for (;;) ;
116  break;
117 
118  } // End switch state
119 
120  runs++; // Increment counter for debugging
121 
122  // No matter the state, wait for approximately a millisecond before we
123  // run the loop again. This gives lower priority tasks a chance to run
124  delay_ms (1);
125  }
126 }

129 /** @brief This method calculates the PWM length to be sent to the servo timer.
130  * @details This method takes a single input of the desired servo position. From this
131  * value, the timer pulse length that corresponds to this position is calculated and
132  * returned.
133  * @return pulse_length A uint8_t type variable that represents the counter value to
134  * be written to timer register so that the correct motor speed will be achieved.
135  */
136 uint8_t task_steering::calc_pwm (int8_t pwm)
137 {
138  // Make sure input is between -90 and 90 degrees
139  if (pwm < -90)
140  {
141  pwm = -90;
142  } else if (pwm > 90) {
143  pwm = 90;
144  }
145 
146  // Convert degrees to pulse length (16-31)
147  return (uint8_t) (24 + (pwm * 0.077));
148 }
