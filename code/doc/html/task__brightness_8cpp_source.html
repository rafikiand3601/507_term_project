67 void task_brightness::run (void)
68 {
69  // Make a variable which will hold times to use for precise task scheduling
70  TickType_t previousTicks = xTaskGetTickCount ();
71 
72  // Create an analog to digital converter driver object and a variable in which to
73  // store its output. The variable p_my_adc only exists within this run() method,
74  // so the A/D converter cannot be used from any other function or method
75  adc* p_my_adc = new adc (p_serial);
76 
77  // Configure counter/timer 3 as a PWM for LED brightness. First set the data
78  // direction register so that the pin used for the PWM will be an output. The
79  // pin is Port E pin 4, which is also OC3B (Output Compare B for Timer 3)
80  DDRE = (1 << 4);
81 
82  // To set 8-bit fast PWM mode we must set bits WGM30 and WGM32, which are in two
83  // different registers (ugh). We use COM3B1 and Com3B0 to set up the PWM so that
84  // the pin output will have inverted sense, that is, a 0 is on and a 1 is off;
85  // this is needed because the LED connects from Vcc to the pin.
86  TCCR3A = (1 << WGM30)
87  | (1 << COM3B1) | (1 << COM3B0);
88 
89  // The CS31 and CS30 bits set the prescaler for this timer/counter to run the
90  // timer at F_CPU / 64
91  TCCR3B = (1 << WGM32)
92  | (1 << CS31) | (1 << CS30);
93 
94  // This is the task loop for the brightness control task. This loop runs until the
95  // power is turned off or something equally dramatic occurs
96  for (;;)
97  {
98  // Read the A/D converter
99  uint16_t a2d_reading = p_my_adc->read_once (0);
100 
101  // Convert the A/D reading into a PWM duty cycle. The A/D reading is between 0
102  // and 1023; the duty cycle should be between 0 and 255. Thus, divide by 4
103  uint16_t duty_cycle = a2d_reading / 4;
104 
105  // Set the brightness. Since the PWM has already been set up, we only need to
106  // put a new value into the duty cycle control register, which on an AVR is
107  // the output compare register for a given timer/counter
108  OCR3B = duty_cycle;
109 
110  // Increment the run counter. This counter belongs to the parent class and can
111  // be printed out for debugging purposes
112  runs++;
113 
114  // This is a method we use to cause a task to make one run through its task
115  // loop every N milliseconds and let other tasks run at other times
116  delay_from_for_ms (previousTicks, 100);
117  }
118 }
