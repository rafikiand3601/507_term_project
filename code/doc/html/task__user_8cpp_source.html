68 void task_user::run (void)
69 {
70  char char_in; // Character read from serial device
71  time_stamp a_time; // Holds the time so it can be displayed
72  uint32_t number_entered = 0; // Holds a number being entered by user
73 
74  // Tell the user how to get into command mode (state 1), where the user interface
75  // task does interesting things such as diagnostic printouts
76  *p_serial << PMS ("Press 'h' or '?' for help") << endl;
77 
78  // This is an infinite loop; it runs until the power is turned off. There is one
79  // such loop inside the code for each task
80  for (;;)
81  {
82  // Run the finite state machine. The variable 'state' is kept by parent class
83  switch (state)
84  {
85  // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
86  // In state 0, we're in command mode, so when the user types characters,
87  // the characters are interpreted as commands to do something
88  case (0):
89  if (p_serial->check_for_char ()) // If the user typed a
90  { // character, read
91  char_in = p_serial->getchar (); // the character
92 
93  // In this switch statement, we respond to different characters as
94  // commands typed in by the user
95  switch (char_in)
96  {
97  case ('o'):
98  *p_serial << PMS ("Object Avoidance") << endl;
99  p_drive_state->put (0x02);
100  break;
101  // The 'r' command starts the car
102  case ('r'):
103  *p_serial << PMS ("run car") << endl;
104  p_drive_state->put (1);
105  break;
106 
107  // The 's' command stops the car from running
108  case ('s'):
109  *p_serial << PMS ("stopping car") << endl;
110  p_drive_state->put (0);
111  break;
112 
113 
114  // The 't' command asks what time it is right now
115  case ('t'):
116  *p_serial << (a_time.set_to_now ()) << endl;
117  break;
118 
119  // The 'v' command asks for version and status information
120  case ('v'):
121  show_status ();
122  break;
123 
124  // The 'd' command has all the tasks dump their stacks
125  case ('d'):
126  print_task_stacks (p_serial);
127  break;
128 
129  // The 'h' command is a plea for help; '?' works also
130  case ('h'):
131  case ('?'):
132  print_help_message ();
133  break;
134 
135  // The 'n' command runs a test of entering a number
136  case ('n'):
137  *p_serial << PMS ("Enter decimal numeric digits, "
138  "then RETURN or ESC") << endl;
139  number_entered = 0;
140  transition_to (1);
141  break;
142 
143  // The 'p' command pings the tracked ball
144  case ('p'):
145  *p_serial << PMS ("sending ping") << endl;
146  p_rf_ping->put (1);
147  break;
148 
149  // A control-C character causes the CPU to restart
150  case (3):
151  *p_serial << PMS ("Resetting AVR") << endl;
152  wdt_enable (WDTO_120MS);
153  for (;;) ;
154  break;
155 
156  // If character isn't recognized, ask What's That Function?
157  default:
158  *p_serial << '"' << char_in << PMS ("\": WTF?") << endl;
159  break;
160  }; // End switch for characters
161  } // End if a character was received
162 
163  break; // End of state 0
164 
165  // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
166  // In state 1, wait for user to enter digits and build 'em into a number
167  case (1):
168  if (p_serial->check_for_char ()) // If the user typed a
169  { // character, read
170  char_in = p_serial->getchar (); // the character
171 
172  // Respond to numeric characters, Enter or Esc only. Numbers are
173  // put into the numeric value we're building up
174  if (char_in >= '0' && char_in <= '9')
175  {
176  *p_serial << char_in;
177  number_entered *= 10;
178  number_entered += char_in - '0';
179  }
180  // Carriage return is ignored; the newline character ends the entry
181  else if (char_in == 10)
182  {
183  *p_serial << "\r";
184  }
185  // Carriage return or Escape ends numeric entry
186  else if (char_in == 13 || char_in == 27)
187  {
188  *p_serial << endl << PMS ("Number entered: ")
189  << number_entered << endl;
190  transition_to (0);
191  }
192  else
193  {
194  *p_serial << PMS ("<invalid char \"") << char_in
195  << PMS ("\">");
196  }
197  }
198 
199  // Check the print queue to see if another task has sent this task
200  // something to be printed
201  else if (p_print_ser_queue->check_for_char ())
202  {
203  p_serial->putchar (p_print_ser_queue->getchar ());
204  }
205 
206  break; // End of state 1
207 
208  // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
209  // We should never get to the default state. If we do, complain and restart
210  default:
211  *p_serial << PMS ("Illegal state! Resetting AVR") << endl;
212  wdt_enable (WDTO_120MS);
213  for (;;) ;
214  break;
215 
216  } // End switch state
217 
218  runs++; // Increment counter for debugging
219 
220  // No matter the state, wait for approximately a millisecond before we
221  // run the loop again. This gives lower priority tasks a chance to run
222  delay_ms (1);
223  }
224 }
