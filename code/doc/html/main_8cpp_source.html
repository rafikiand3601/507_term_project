201 /** @brief An ISR routine to determine the rotational velocity of the motor.
202  * @details This routine is used to determine the rotational velocity of the motor.
203  * While it runs, it has not been fully implemented in our code.
204  */
205 ISR(TIMER3_CAPT_vect)
206 {
207  uint16_t count1 = TCNT3;
208  //width_1->ISR_put(1); //store value of pulse width
209  if (edge_1->ISR_get ()) // rising edge
210  {
211  // set Edge Sense set to falling edge, clear counter
212  TCCR3B &= ~(1 << ICES3);
213  TCNT3 = 0;
214  edge_1->ISR_put(0); //Toggle edge_1 to 0
215  }
216  else // falling edge
217  {
218  // set Edge Sense set to rising edge
219  TCCR3B |= (1 << ICES3);
220  width_1->ISR_put(count1); //store value of pulse width
221  edge_1->ISR_put(1); //Toggle edge_1 to 1
