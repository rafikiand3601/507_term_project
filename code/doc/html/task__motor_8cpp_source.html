48 void task_motor::run (void)
49 {
50 
51  offset = -2;
52 
53  // This is an infinite loop; it runs until the power is turned off. There is one
54  // such loop inside the code for each task
55  for (;;)
56  {
57  // Run the finite state machine. The variable 'state' is kept by parent class
58  switch (state)
59  {
60  // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
61  // In state 0, setup the output pin for the motor on PB5(OC1A)
62  case (0):
63  // Need a wave with period ~ 20 ms and pulse length of 1.0-2.0 ms
64  /** f = f_clk / N*(1 + TOP)
65  * Want f = 50 [hz]
66  * TOP = 0xFF = 255, f_clk = 16 [Mhz]
67  * N = 1024 ---> f = 61.3 [hz] close enough
68  */
69 
70  // Set PB5 as output pin
71  DDRB |= (1 << PB5);
72  // Setup register for fast pwm, non-inverting
73  // WGM: fast pwm 0x03FF/1023 COM1A1: non-inverting output
74  TCCR1A |= (1 << WGM11) | (1 << COM1A1);
75  // WGM: fast pwm 0x03FF CS: prescaler = 256
76  TCCR1B |= (1 << WGM12) | (1 << WGM13) | (1 << CS12);
77  // TCCR1C unused
78 
79  // ICR = 0xFFFF counts to this value
80  ICR1H = 0x04;
81  ICR1L = 0xE1;
82 
83  // Setup of OCR
84  // 1.5 [ms] ----> 667 [hz] ---> 94
85  // Should run 50.0 [hz] period with 1.5 [ms] pulses
86  OCR1AH = 0x00;
87  OCR1AL = 0x5E;
88 
89  // Move to control state
90  state = 1;
91  break; // End of state 0
92 
93  // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
94  // In state 1, control the esc power
95  case (1):
96  // Vary OCR to change pulse length.
97  // Pulses are between 1.0 and 2.0 [ms]
98 
99  OCR1AL = calc_pwm(p_motor_vel->get ());
100 
101  break; // End of state 1
102 
103  // - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
104  // We should never get to the default state. If we do, complain and restart
105  default:
106  *p_serial << PMS ("Illegal state! Resetting AVR") << endl;
107  wdt_enable (WDTO_120MS);
108  for (;;) ;
109  break;
110 
111  } // End switch state
112 
113  runs++; // Increment counter for debugging
114 
115  // No matter the state, wait for approximately a millisecond before we
116  // run the loop again. This gives lower priority tasks a chance to run
117  delay_ms (1);
118  }
